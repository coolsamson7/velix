SourceNode? findMatchingNode(Match match, int index) {
  if (index < match.paths[0].length) {
    for (var child in children) {
      if (child.accessor == match.paths[0][index]) {
        return child.findMatchingNode(match, index + 1);
      }
    } // for
  } // if

  return this;
}
